import { Meta, ColorPalette, ColorItem } from '@storybook/addon-docs/blocks';
import {lightTheme, darkTheme} from '../lib/colors';
 
<Meta title="Colors" />
 
<h2>Light Theme</h2>
{Object.entries(lightTheme).map(([name, value]) => (
  <ColorPalette key={name}>
  <ColorItem
    key={name}
    title={name}
    subtitle={value.description}
    colors={{ [name]: value}}
  />
  </ColorPalette>
))}
<h2>Dark Theme</h2>
{Object.entries(darkTheme).map(([name, value]) => (
  <ColorPalette key={name}>
  <ColorItem
    key={name}
    title={name}
    subtitle={value.description}
    colors={{ [name]: value}}
  />
  </ColorPalette>
))}
